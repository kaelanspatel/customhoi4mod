void replacef_flip(int dim, pixel* src, pixel* dst){
    int i, j, replace, p;
    int k = 0;
    for (i = 0; i< dim; i+=4){
        p = 0;
        for(j = 0; j<dim; j++){
            dst[p + (dim-1-i)] = src[k + j];
            dst[dim*(j+1) + (dim-1-i)] = src[k + j +1];
            dst[dim*(j+2) + (dim-1-i)] = src[k + j + 2];
            dst[dim*(j+3) + (dim-1-i)] = src[k + j + 3];
            p += dim;
        }
        k += dim *4;
    }
}
